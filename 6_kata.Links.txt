#include<vector>

std::vector<unsigned long long> partsSum(const std::vector<unsigned long long>& ls)
{
	//sum_results для хранения результатов сумм
	//для хранения суммы всего вектора
	std::vector<unsigned long long> sum_results;
	unsigned long long total_sum = 0;

	//цикл, который вычисляет сумму всего вектора
	for (const auto& num : ls)
	{
		total_sum += num;
	}

	//пушим сумму всех элементов вектора в вектор sum_results
	sum_results.push_back(total_sum);

	//цикл, который проходится по всему вектору ls, 
	//каждую итерацию цикла вычитая из общей суммы текущий элемент массива 
	//и пушит результат суммы в вектор sum_results
	for (size_t i = 0; i < ls.size(); ++i)
	{
		total_sum -= ls[i];
		sum_results.push_back(total_sum);
	}

	return sum_results;

	//ls = [1, 2, 3, 4, 5, 6]

	//[21, 18, 15, 11, 6, 0

	//21 - 1 = 20
	//20 - 2 = 18
	//18 - 3 = 15
	//15 - 4 = 11
	//11 - 5 = 6
	//6 - 6 = 0
}
