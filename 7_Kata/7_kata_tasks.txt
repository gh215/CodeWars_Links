int save(std::vector<int> sizes, int hd) 
{
  int free_space = 0;
  int result = 0;
  for (size_t i = 0; i < sizes.size(); i++)    
  {
    if (free_space + sizes[i] > hd)
    {
      break;
    }
    else
    {
      free_space += sizes[i];
      result++;
    }
  }
  
  return result;
}
